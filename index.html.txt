  <div id="menu">
    <h1>üêî Franguinho Dash</h1>

  <canvas id="game" width="350" height="500"></canvas>

  <script>
    const canvas = document.getElementById("game");
    const ctx = canvas.getContext("2d");
    const info = document.getElementById("info");
    const menu = document.getElementById("menu");
    const startBtn = document.getElementById("startBtn");

    let frangoImg = new Image();
    // Imagem hospedada online. Se quiser usar local, troque para 'frango.png' e adicione ao ZIP.
    frangoImg.src = "https://i.imgur.com/7kCEHoE.png";

    let frango = { x: 150, y: 400, size: 50, speed: 6 };
    let objetos = [];
    let moedas = [];
    let score = 0;
    let desconto = 0;
    let playing = false;

    function desenharFrango() {
      if (frangoImg.complete && frangoImg.naturalWidth !== 0) {
        ctx.drawImage(frangoImg, frango.x, frango.y, frango.size, frango.size);
      } else {
        ctx.fillStyle = "#ffa500";
        ctx.beginPath();
        ctx.arc(frango.x + frango.size/2, frango.y + frango.size/2, frango.size/2, 0, Math.PI*2);
        ctx.fill();
      }
    }

    function criarObjeto() {
      objetos.push({ x: Math.random() * (canvas.width - 30), y: -40, size: 30, speed: 3 + Math.random()*2 });
    }

    function criarMoeda() {
      moedas.push({ x: Math.random() * (canvas.width - 20), y: -30, size: 20, speed: 2 + Math.random()*1.5 });
    }

    function desenharObjetos() {
      ctx.fillStyle = "#e74c3c";
      objetos.forEach(o => ctx.fillRect(o.x, o.y, o.size, o.size));
    }

    function desenharMoedas() {
      ctx.fillStyle = "gold";
      moedas.forEach(m => {
        ctx.beginPath();
        ctx.arc(m.x + m.size/2, m.y + m.size/2, m.size/2, 0, Math.PI * 2);
        ctx.fill();
        ctx.strokeStyle = "#b8860b";
        ctx.lineWidth = 1;
        ctx.stroke();
      });
    }

    function atualizar() {
      if (!playing) return;
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      if (Math.random() < 0.03) criarObjeto();
      if (Math.random() < 0.02) criarMoeda();

      objetos.forEach(o => o.y += o.speed);
      moedas.forEach(m => m.y += m.speed);

      objetos = objetos.filter(o => o.y < canvas.height + 50);
      moedas = moedas.filter(m => m.y < canvas.height + 50);

      for (let o of objetos) {
        if (
          frango.x < o.x + o.size &&
          frango.x + frango.size > o.x &&
          frango.y < o.y + o.size &&
          frango.y + frango.size > o.y
        ) {
          playing = false;
          setTimeout(()=> {
            alert("O franguinho foi atingido!\\nMoedas coletadas: " + score + "\\nDesconto aplicado: R$" + desconto.toFixed(2));
            location.reload();
          }, 50);
          return;
        }
      }

      for (let i = moedas.length -1; i >=0; i--) {
        const m = moedas[i];
        if (
          frango.x < m.x + m.size &&
          frango.x + frango.size > m.x &&
          frango.y < m.y + m.size &&
          frango.y + frango.size > m.y
        ) {
          score++;
          desconto = score * 0.10;
          moedas.splice(i,1);
          info.textContent = `Moedas: ${score} | Desconto: R$${desconto.toFixed(2)}`;
        }
      }

      desenharFrango();
      desenharObjetos();
      desenharMoedas();

      requestAnimationFrame(atualizar);
    }

    document.addEventListener("keydown", e => {
      if (!playing) return;
      if (e.key === "ArrowLeft" || e.key === "a") frango.x = Math.max(0, frango.x - frango.speed);
      if (e.key === "ArrowRight" || e.key === "d") frango.x = Math.min(canvas.width - frango.size, frango.x + frango.speed);
    });

    // Toque / arrastar para celular
    let touchStartX = null;
    canvas.addEventListener("touchstart", (e) => {
      touchStartX = e.touches[0].clientX;
    });
    canvas.addEventListener("touchmove", (e) => {
      if (!playing) return;
      const touchX = e.touches[0].clientX;
      const rect = canvas.getBoundingClientRect();
      let relativeX = touchX - rect.left - frango.size/2;
      frango.x = Math.max(0, Math.min(canvas.width - frango.size, relativeX));
      e.preventDefault();
    }, {passive:false});

    startBtn.onclick = () => {
      menu.style.display = "none";
      canvas.style.display = "block";
      playing = true;
      atualizar();
    }
  </script>
